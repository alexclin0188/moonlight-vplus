### 核心接口

```kotlin
class StreamEngine(private val activity: Activity) {
    // 配置
    val prefConfig: PreferenceConfiguration

    // 连接相关
    var conn: NvConnection?           // 串流连接
    var app: NvApp                    // 当前串流应用

    // 控制器相关
    var controllerHandler: ControllerHandler?
    var controllerManager: ControllerManager?

    // 功能开关状态
    var isCrownFeatureEnabled: Boolean  // 按键映射（原Crown）
    var isAudioMuted: Boolean           // 音频静音
    var isTouchOverrideEnabled: Boolean // 平移缩放

    // ── 初始化 ──
    fun initialize(intent: Intent): Boolean
        // 1. 解析 Intent extras（10个字段）
        // 2. PreferenceConfiguration.readPreferences(activity)
        // 3. 构建 StreamConfiguration
        // 4. 创建 ControllerHandler + AudioRenderer
        // 5. 创建 NvConnection
        // 6. 条件创建 ControllerManager

    // ── 启动 ──
    fun startStreaming(surface: Surface, width: Int, height: Int)
        // 调用 conn.start() + 解码器渲染

    // ── 断开 ──
    fun disconnect()                   // finish Activity
    fun disconnectAndQuit()             // finish + conn.doStopAndQuit()

    // ── 各功能toggle ──
    fun changeResolution()             // 切换分辨率并recreate
    fun toggleKeyboard()               // Android软键盘切换
    fun toggleVirtualKeyboard()        // 虚拟键盘覆盖层
    fun toggleMicrophoneButton()       // 麦克风
    fun toggleAudioMute()              // 静音
    fun togglePerformanceOverlay()     // 性能图层
    fun toggleVirtualController()      // 虚拟手柄

    // ── 生命周期 ──
    fun onResume()
    fun onPause()
    fun release()                      // 全资源释放
}
```

### 关键初始化步骤（按 Game.kt 中顺序）

1. `PreferenceConfiguration.readPreferences(activity)` — 读取所有SharedPreferences
2. `NvApp(appName, appId, hdrSupported)` — 创建应用对象
3. 构建 `StreamConfiguration.Builder`:
   - width / height / fps / bitrate
   - videoFormat / enableHdr / hdrMode
   - audioConfig / audioCodec / enableAudioPassthrough
   - enablePerfOverlay / perfOverlayLocked
   - framePacing / abrMode / enableAdaptiveBitrate
   - 等约 30 个参数
4. 创建 `ControllerHandler`
5. 创建 AudioRenderer
6. 创建 `NvConnection(host, port, app, streamConfig, cryptoProvider, ...)`
7. 条件创建 `ControllerManager`（若 `onscreenKeyboard` 开启）

### 注意事项

- 此阶段只实现**最小可用**的初始化：不要求所有 toggle 方法立即可用
- conn.start() 的回调 listener 仅做日志记录即可
- 后续阶段逐步补充功能

### onCreate 流程

```kotlin
override fun onCreate(savedInstanceState: Bundle?) {
    super.onCreate(savedInstanceState)
    
    // 1. 解析 Intent
    val intent = intent
    
    // 2. 初始化引擎
    engine = StreamEngine(this)
    if (!engine.initialize(intent)) {
        Toast.makeText(this, "串流初始化失败", Toast.LENGTH_SHORT).show()
        finish()
        return
    }
    
    // 3. 设置 Content：SurfaceView + Compose Overlay
    setContent {
        val context = LocalContext.current
        val prefs = remember { PreferenceManager.getDefaultSharedPreferences(context) }
        val darkTheme = when (prefs.getString("list_theme_mode", "dark")) {
            "dark" -> true
            "light" -> false
            else -> isSystemInDarkTheme()
        }
        
        MoonLinkTheme(darkTheme = darkTheme) {
            Box(modifier = Modifier.fillMaxSize()) {
                // 串流画面（SurfaceView）
                AndroidView(
                    factory = { ctx ->
                        SurfaceView(ctx).also { sv ->
                            surfaceView = sv
                            sv.holder.addCallback(surfaceCallback)
                        }
                    },
                    modifier = Modifier.fillMaxSize()
                )
                
                // 面板 overlay（后续阶段实现，此阶段为空）
                StreamOverlay(engine = engine)
            }
        }
    }
}
```

## 0.4 入口注入（alexclin 包 → 直接跳新页面）

> **原则**：`com.alexclin.moonlink` 包下的入口始终跳转 `StreamActivity`；`com.limelight.PcView`（旧主页）保持不变，继续跳转旧 `Game.kt`。无需 feature flag。

### DeviceOverviewScreen.kt 修改

在 `launchStreamFromOverview()` 函数中（约行1051-1086），将 Intent 创建改为：

```kotlin
// 新逻辑（alexclin 包入口）：
// 仅当 useLastSettings 为 true 时读取 AppSettingsManager 的配置并写入 prefConfig，
// 然后统一构造跳转 StreamActivity 的 Intent

val intent = if (useLastSettings) {
    // 先让 AppSettingsManager 将保存的配置写入 PreferenceConfiguration
    appSettingsManager.applyLastSettingsToPreferences(computer)
    createStreamIntent(context, computer, targetApp, managerBinder, forceResume)
} else {
    createStreamIntent(context, computer, targetApp, managerBinder, forceResume)
}
context.startActivity(intent)

// 辅助函数
private fun createStreamIntent(
    context: Context,
    computer: ComputerDetails,
    app: NvApp,
    managerBinder: ComputerManagerService.ComputerManagerBinder,
    forceResume: Boolean = false
): Intent {
    return Intent(context, StreamActivity::class.java).apply {
        putExtra("Host", computer.activeAddress?.address)
        putExtra("Port", computer.activeAddress?.port ?: 0)
        putExtra("HttpsPort", computer.httpsPort)
        putExtra("AppName", app.appName)
        putExtra("AppId", app.appId)
        putExtra("AppHdr", app.hdrSupported)
        putExtra("PcUuid", computer.uuid)
        putExtra("PcName", computer.name)
        putExtra("UniqueId", managerBinder.uniqueId)
        app.cmdList?.let { putExtra("AppCmd", it.toString()) }
        computer.serverCert?.let { putExtra("ServerCert", it.encoded) }
        putExtra("forceResume", forceResume)
        computer.activeAddress?.let {
            putExtra("Address", "${it.address}:${it.port}")
        }
    }
}
```

### DeviceListScreen 同样处理

在设备列表点击缩略图跳转串流的位置，直接用 `createStreamIntent()`。

> 旧入口 `com.limelight.PcView` 不走此路径，始终调用 `ServerHelper.createStartIntent()` → 旧 `Game.kt`。
