## 4.2 TouchModeSection.kt — 触控模式 4 选 1

**路径**：`app/src/main/java/com/alexclin/moonlink/stream/ui/panels/TouchModeSection.kt`

### 4 种模式

| 模式 | 标志组合 |
|------|----------|
| 增强式多点触控 | `enableEnhancedTouch=true`, `touchscreenTrackpad=false`, `nativeMousePointer=false` |
| 经典鼠标模式 | 三者均为 `false` |
| 触控板模式 | `touchscreenTrackpad=true`, `nativeMousePointer=false` |
| 本地鼠标指针 | `nativeMousePointer=true` |

### 实现

```kotlin
enum class TouchMode(val label: String) {
    ENHANCED("增强式多点触控"),
    CLASSIC("经典鼠标模式"),
    TRACKPAD("触控板模式"),
    NATIVE_MOUSE("本地鼠标指针")
}

@Composable
fun TouchModeSection(engine: StreamEngine) {
    val currentMode = remember(engine.prefConfig) {
        when {
            engine.prefConfig.enableNativeMousePointer -> TouchMode.NATIVE_MOUSE
            engine.prefConfig.touchscreenTrackpad -> TouchMode.TRACKPAD
            engine.prefConfig.enableEnhancedTouch -> TouchMode.ENHANCED
            else -> TouchMode.CLASSIC
        }
    }
    var selectedMode by remember { mutableStateOf(currentMode) }

    Column(modifier = Modifier.padding(horizontal = 12.dp, vertical = 8.dp)) {
        Text("触控模式", style = MaterialTheme.typography.titleSmall, color = MaterialTheme.colorScheme.onSurface)
        Spacer(modifier = Modifier.height(6.dp))

        // 4 个 Chip 横向排列，可滚动
        LazyRow(horizontalArrangement = Arrangement.spacedBy(6.dp)) {
            items(TouchMode.entries.toList()) { mode ->
                FilterChip(
                    selected = selectedMode == mode,
                    onClick = {
                        selectedMode = mode
                        applyTouchMode(engine, mode)
                    },
                    label = { Text(mode.label, style = MaterialTheme.typography.labelMedium) }
                )
            }
        }

        // 关联子项
        AnimatedVisibility(visible = selectedMode == TouchMode.CLASSIC) {
            // 显示/隐藏远程鼠标按钮
            TextButton(onClick = { KeySender.sendRemoteMouseToggle(engine.conn) }) {
                Text("显示/隐藏远程鼠标")
            }
        }
        AnimatedVisibility(visible = selectedMode == TouchMode.TRACKPAD) {
            Column {
                var doubleClickDrag by remember { mutableStateOf(engine.prefConfig.enableDoubleClickDrag) }
                Row {
                    Text("双击按住", modifier = Modifier.weight(1f))
                    Switch(
                        checked = doubleClickDrag,
                        onCheckedChange = { doubleClickDrag = it; engine.prefConfig.enableDoubleClickDrag = it; engine.prefConfig.writePreferences(engine.activity) }
                    )
                }
                var localCursor by remember { mutableStateOf(engine.prefConfig.enableLocalCursorRendering) }
                Row {
                    Text("本地光标渲染", modifier = Modifier.weight(1f))
                    Switch(
                        checked = localCursor,
                        onCheckedChange = { localCursor = it; engine.prefConfig.enableLocalCursorRendering = it; engine.prefConfig.writePreferences(engine.activity) }
                    )
                }
            }
        }
    }
}

private fun applyTouchMode(engine: StreamEngine, mode: TouchMode) {
    val ctx = engine.activity
    when (mode) {
        TouchMode.ENHANCED -> {
            engine.prefConfig.enableEnhancedTouch = true
            engine.prefConfig.enableNativeMousePointer = false
            engine.prefConfig.touchscreenTrackpad = false
            Toast.makeText(ctx, "已切换为增强式多点触控", Toast.LENGTH_SHORT).show()
        }
        TouchMode.CLASSIC -> {
            engine.prefConfig.enableEnhancedTouch = false
            engine.prefConfig.enableNativeMousePointer = false
            engine.prefConfig.touchscreenTrackpad = false
            Toast.makeText(ctx, "已切换为经典鼠标模式", Toast.LENGTH_SHORT).show()
        }
        TouchMode.TRACKPAD -> {
            engine.prefConfig.enableEnhancedTouch = false
            engine.prefConfig.enableNativeMousePointer = false
            engine.prefConfig.touchscreenTrackpad = true
            Toast.makeText(ctx, "已切换为触控板模式", Toast.LENGTH_SHORT).show()
        }
        TouchMode.NATIVE_MOUSE -> {
            engine.prefConfig.enableEnhancedTouch = false
            engine.prefConfig.enableNativeMousePointer = true
            engine.prefConfig.touchscreenTrackpad = false
            Toast.makeText(ctx, "已切换为本地鼠标指针", Toast.LENGTH_SHORT).show()
        }
    }
    engine.prefConfig.writePreferences(ctx)
}
```

```kotlin
@Composable
fun KeyMappingSection(engine: StreamEngine) {
    var enabled by remember { mutableStateOf(engine.isCrownFeatureEnabled) }

    Column(modifier = Modifier.padding(horizontal = 12.dp, vertical = 8.dp)) {
        // 顶层开关
        Row(verticalAlignment = Alignment.CenterVertically) {
            Icon(Icons.Default.TextFields, contentDescription = null, modifier = Modifier.size(20.dp), tint = MaterialTheme.colorScheme.primary)
            Spacer(modifier = Modifier.width(8.dp))
            Text("启用按键映射", style = MaterialTheme.typography.bodyMedium, modifier = Modifier.weight(1f))
            Switch(
                checked = enabled,
                onCheckedChange = {
                    enabled = it
                    engine.isCrownFeatureEnabled = it
                }
            )
        }

        // 开启后展示三个子入口
        AnimatedVisibility(visible = enabled) {
            Column(modifier = Modifier.padding(start = 28.dp)) {
                // 1. 切换按键映射方案
                TextButton(
                    onClick = {
                        engine.controllerManager?.pageConfigController?.open()
                    },
                    modifier = Modifier.fillMaxWidth()
                ) {
                    Text("切换按键映射方案 >", style = MaterialTheme.typography.bodyMedium)
                }

                // 2. 编辑当前方案
                TextButton(
                    onClick = {
                        engine.controllerManager?.elementController?.changeMode(ElementController.Mode.Edit)
                        engine.controllerManager?.elementController?.open()
                    },
                    modifier = Modifier.fillMaxWidth()
                ) {
                    Text("编辑当前方案 >", style = MaterialTheme.typography.bodyMedium)
                }

                // 3. 其它设置
                var showOtherSettings by remember { mutableStateOf(false) }
                TextButton(
                    onClick = { showOtherSettings = !showOtherSettings },
                    modifier = Modifier.fillMaxWidth()
                ) {
                    Text("其它设置 ${if (showOtherSettings) "▲" else "▼"}",
                         style = MaterialTheme.typography.bodyMedium)
                }
                AnimatedVisibility(visible = showOtherSettings) {
                    KeyMappingOtherSettings(engine = engine)
                }
            }
        }
    }
}
```

## 4.4 DisplaySection.kt — 显示详情

**路径**：`app/src/main/java/com/alexclin/moonlink/stream/ui/detail/DisplaySection.kt`

### 子项列表

1. **帧率切换** — `list_fps`
2. **码率切换** — `seekbar_bitrate_kbps`（分段滑块）
3. **多显示屏管理** — 主机显示器 + VDD + 本地外接显示器 + 分辨率缩放
4. **自适应流畅度** — `checkbox_adaptive_bitrate` + 子项
5. **HDR 开关** — `checkbox_enable_hdr` + 子项
6. **画面设置** — 拉伸/反转/旋转/位置/偏移

### 码率分段滑块

从 `BitrateCardController` 复制 5 段 60 个位置的映射表：

```
位置 0-9:    500 - 5,000 kbps     (步进 500)
位置 10-24:  6,000 - 20,000 kbps  (步进 1,000)
位置 25-39:  22,000 - 50,000 kbps  (步进 2,000)
位置 40-49:  55,000 - 100,000 kbps (步进 5,000)
位置 50-59:  110,000 - 200,000 kbps(步进 10,000)
```

滑块松开后调用 `engine.conn.setBitrate(kbps, callback)` 实时生效。

### 多显示屏管理

```kotlin
@Composable
fun DisplaySelector(engine: StreamEngine) {
    var displays by remember { mutableStateOf<List<NvDisplay>>(emptyList()) }

    LaunchedEffect(Unit) {
        withContext(Dispatchers.IO) {
            try {
                displays = engine.conn?.getDisplays() ?: emptyList()
            } catch (_: Exception) {}
        }
    }

    // 单选列表
    displays.forEach { display ->
        Row {
            RadioButton(...)
            Text(display.friendlyName)
        }
    }
    // VDD 开关
    Row {
        Text("虚拟显示器 (VDD)")
        Switch(checked = engine.prefConfig.useVdd, onCheckedChange = { ... })
    }
    // 本地外接显示器
    Row {
        Text("使用外接显示器")
        Switch(checked = engine.prefConfig.useExternalDisplay, onCheckedChange = { ... })
    }
}
```
