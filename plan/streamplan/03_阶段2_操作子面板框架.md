### 实现

```kotlin
enum class DetailPage {
    MAIN_LIST,          // 主列表
    DISPLAY,            // 显示详情
    HOST_SETTINGS,      // 主机设置详情
    SHORTCUT_ACTIONS,   // 快捷操作详情
    PERIPHERALS,        // 外设详情
    KEY_MAPPING         // 按键映射方案选择（独立全屏页面）
}

@Composable
fun SubPanelContainer(engine: StreamEngine) {
    var detailPage by remember { mutableStateOf(DetailPage.MAIN_LIST) }
    val configuration = LocalConfiguration.current
    val panelWidth = (configuration.screenWidthDp.dp * 0.45f).coerceIn(280.dp, 400.dp)

    Surface(
        modifier = Modifier
            .align(Alignment.CenterEnd)  // 紧贴右边缘
            .fillMaxHeight()
            .padding(end = 60.dp)  // 留出竖向窄条宽度
            .width(panelWidth),
        shape = RoundedCornerShape(topStart = 16.dp, bottomStart = 16.dp),
        color = MaterialTheme.colorScheme.surface,
        shadowElevation = 8.dp
    ) {
        AnimatedContent(
            targetState = detailPage,
            transitionSpec = {
                val direction = if (targetState == DetailPage.MAIN_LIST) {
                    // 返回列表：从左侧滑入
                    ContentTransform(
                        targetContentEnter = slideInHorizontally(
                            initialOffsetX = { -it / 3 },
                            animationSpec = tween(250, easing = FastOutSlowInEasing)
                        ) + fadeIn(animationSpec = tween(200)),
                        initialContentExit = slideOutHorizontally(
                            targetOffsetX = { it },
                            animationSpec = tween(250, easing = FastOutSlowInEasing)
                        ) + fadeOut(animationSpec = tween(200))
                    )
                } else {
                    // 进入详情：从右侧滑入
                    ContentTransform(
                        targetContentEnter = slideInHorizontally(
                            initialOffsetX = { it },
                            animationSpec = tween(250, easing = FastOutSlowInEasing)
                        ) + fadeIn(animationSpec = tween(200)),
                        initialContentExit = slideOutHorizontally(
                            targetOffsetX = { -it / 3 },
                            animationSpec = tween(250, easing = FastOutSlowInEasing)
                        ) + fadeOut(animationSpec = tween(200))
                    )
                }
            }
        ) { page ->
            when (page) {
                DetailPage.MAIN_LIST -> {
                    MainListView(
                        engine = engine,
                        onNavigate = { detailPage = it }
                    )
                }
                DetailPage.DISPLAY -> {
                    DisplaySection(
                        engine = engine,
                        onBack = { detailPage = DetailPage.MAIN_LIST }
                    )
                }
                DetailPage.HOST_SETTINGS -> {
                    HostSettingsSection(
                        engine = engine,
                        onBack = { detailPage = DetailPage.MAIN_LIST }
                    )
                }
                DetailPage.SHORTCUT_ACTIONS -> {
                    ShortcutActionsSection(
                        engine = engine,
                        onBack = { detailPage = DetailPage.MAIN_LIST }
                    )
                }
                DetailPage.PERIPHERALS -> {
                    PeripheralsDetail(
                        engine = engine,
                        onBack = { detailPage = DetailPage.MAIN_LIST }
                    )
                }
            }
        }
    }
}
```

## 2.2 MainListView — 主列表

**路径**：内嵌于 `SubPanelContainer.kt` 或独立文件

```kotlin
@Composable
private fun MainListView(
    engine: StreamEngine,
    onNavigate: (DetailPage) -> Unit
) {
    LazyColumn(
        modifier = Modifier.fillMaxSize(),
        contentPadding = PaddingValues(horizontal = 12.dp, vertical = 8.dp)
    ) {
        // ── 1. 快捷操作行 ──
        item { QuickActionRow(engine = engine) }
        item { Divider(modifier = Modifier.padding(vertical = 4.dp)) }

        // ── 2. 平移缩放 ──
        if (engine.prefConfig.enablePip) {
            item { PanZoomSection(engine = engine) }
            item { Divider(modifier = Modifier.padding(vertical = 4.dp)) }
        }

        // ── 3. 启用按键映射 ──
        item { KeyMappingSection(engine = engine) }
        item { Divider(modifier = Modifier.padding(vertical = 4.dp)) }

        // ── 4. 触控模式 ──
        item { TouchModeSection(engine = engine) }
        item { Divider(modifier = Modifier.padding(vertical = 4.dp)) }

        // ── 5. 显示 > ──
        item { SectionEntryRow(icon = Icons.Default.Tv, label = "显示", onClick = { onNavigate(DetailPage.DISPLAY) }) }

        // ── 6. 主机设置 > ──
        item { SectionEntryRow(icon = Icons.Default.Computer, label = "主机设置", onClick = { onNavigate(DetailPage.HOST_SETTINGS) }) }

        // ── 7. 快捷操作 > ──
        item { SectionEntryRow(icon = Icons.Default.Bolt, label = "快捷操作", onClick = { onNavigate(DetailPage.SHORTCUT_ACTIONS) }) }
        item { Divider(modifier = Modifier.padding(vertical = 4.dp)) }

        // ── 8. 外设 ──
        item { PeripheralsSection(engine = engine, onNavigate = { onNavigate(DetailPage.PERIPHERALS) }) }
        item { Divider(modifier = Modifier.padding(vertical = 4.dp)) }

        // ── 9. 体感助手 ──
        item { GyroSection(engine = engine) }
        item { Divider(modifier = Modifier.padding(vertical = 4.dp)) }

        // ── 10. 更多 ──
        item { MoreSection(engine = engine) }

        // 底部留白
        item { Spacer(modifier = Modifier.height(24.dp)) }
    }
}
```

### 分组入口行组件

```kotlin
@Composable
private fun SectionEntryRow(
    icon: ImageVector,
    label: String,
    onClick: () -> Unit
) {
    Row(
        modifier = Modifier
            .fillMaxWidth()
            .clickable(onClick = onClick)
            .padding(horizontal = 12.dp, vertical = 14.dp),
        verticalAlignment = Alignment.CenterVertically
    ) {
        Icon(
            icon,
            contentDescription = null,
            modifier = Modifier.size(20.dp),
            tint = MaterialTheme.colorScheme.primary
        )
        Spacer(modifier = Modifier.width(12.dp))
        Text(
            label,
            style = MaterialTheme.typography.bodyLarge,
            color = MaterialTheme.colorScheme.onSurface,
            modifier = Modifier.weight(1f)
        )
        Icon(
            Icons.Default.ChevronRight,
            contentDescription = null,
            modifier = Modifier.size(18.dp),
            tint = MaterialTheme.colorScheme.onSurfaceVariant
        )
    }
}
```

## 2.4 产出验证

1. 竖向窄条点击"操作" → 操作子面板从右侧滑入（紧贴窄条左侧）
2. 子面板宽度约 45% 屏幕宽，竖直方向填满屏幕
3. 子面板包含快捷操作行 + 9 个功能分组
4. 点击"显示 >" → 主列表向左滑出，显示详情从右滑入
5. 详情顶部有"← 返回"按钮
6. 点击返回 → 详情向右滑出，主列表从左侧滑入
7. 必须先返回列表才能进入另一个分类（不能从显示详情直接跳到主机设置）
8. 点击面板外侧区域 → 全版面关闭 → 悬浮按钮重新出现
