## 3.1 QuickActionRow.kt

**路径**：`app/src/main/java/com/alexclin/moonlink/stream/ui/panels/QuickActionRow.kt`

### 固定按钮

| ID | 标签 | 图标 | 行为 |
|----|------|------|------|
| `quit` | 退出串流 | `Icons.Default.ExitToApp` | `engine.disconnect()` |
| `quit_and_stop` | 退出并终止 | `Icons.Default.PowerSettingsNew` | `engine.disconnectAndQuit()` |

### 可配置按钮（默认）

| ID | 标签 | 可见条件 |
|----|------|----------|
| `toggle_audio` | 声音 | 始终可见 |
| `toggle_pip` | 画中画 | 始终可见 |
| `toggle_mic` | 麦克风 | 麦克风总开关开启时 |

### 实现

```kotlin
@Composable
fun QuickActionRow(engine: StreamEngine) {
    val context = LocalContext.current
    val configIds = remember { QuickActionRegistry.loadConfig(context).toMutableList() }
    var showEditor by remember { mutableStateOf(false) }
    var editMode by remember { mutableStateOf(false) }

    Column {
        Row(
            modifier = Modifier.fillMaxWidth().padding(horizontal = 8.dp, vertical = 4.dp),
            verticalAlignment = Alignment.CenterVertically
        ) {
            Text("快捷操作", style = MaterialTheme.typography.labelMedium,
                 color = MaterialTheme.colorScheme.onSurfaceVariant)
            Spacer(modifier = Modifier.weight(1f))
            if (editMode) {
                TextButton(onClick = { editMode = false }) { Text("完成", style = MaterialTheme.typography.labelSmall) }
            }
            IconButton(onClick = { showEditor = true }, modifier = Modifier.size(28.dp)) {
                Icon(Icons.Default.Add, contentDescription = "编辑", modifier = Modifier.size(16.dp))
            }
        }

        LazyRow(
            modifier = Modifier.fillMaxWidth(),
            horizontalArrangement = Arrangement.spacedBy(4.dp),
            contentPadding = PaddingValues(horizontal = 8.dp)
        ) {
            // 固定按钮
            item {
                QuickActionChip(
                    label = "退出串流",
                    icon = Icons.Default.ExitToApp,
                    isFixed = true,
                    onClick = { engine.disconnect() }
                )
            }
            item {
                QuickActionChip(
                    label = "退出并终止",
                    icon = Icons.Default.PowerSettingsNew,
                    isFixed = true,
                    onClick = { engine.disconnectAndQuit() }
                )
            }

            // 可配置按钮
            items(configIds.size) { index ->
                val id = configIds[index]
                val action = QuickActionRegistry.getBuiltin(id) ?: return@items
                val label = context.getString(action.labelRes)

                if (id == "toggle_mic" && !engine.prefConfig.enableMic) {
                    // 麦克风总开关关闭时隐藏麦克风按钮
                    return@items
                }

                QuickActionChip(
                    label = label,
                    icon = /* from action.iconRes */,
                    isFixed = false,
                    editMode = editMode,
                    onClick = { executeAction(id, engine) },
                    onDelete = if (editMode) {
                        {
                            configIds.removeAt(index)
                            QuickActionRegistry.saveConfig(context, configIds)
                        }
                    } else null,
                    onDragStart = if (editMode) { ... } else null
                )
            }
        }
    }

    // ── 编辑弹窗 ──
    if (showEditor) {
        QuickActionEditorDialog(
            currentIds = configIds,
            onSave = { newIds ->
                QuickActionRegistry.saveConfig(context, newIds)
                showEditor = false
            },
            onDismiss = { showEditor = false }
        )
    }
}
```

### QuickActionChip 组件

```kotlin
@Composable
private fun QuickActionChip(
    label: String,
    icon: ImageVector?,
    isFixed: Boolean,
    editMode: Boolean = false,
    onClick: () -> Unit,
    onDelete: (() -> Unit)? = null,
    onDragStart: (() -> Unit)? = null
) {
    val scale by animateFloatAsState(if (editMode) 1.05f else 1f)

    Surface(
        modifier = Modifier
            .graphicsLayer { scaleX = scale; scaleY = scale }
            .then(if (onDragStart != null) Modifier.pointerInput(Unit) {
                detectDragGesturesAfterLongPress(
                    onDragStart = { onDragStart() },
                    onDragEnd = {},
                    onDragCancel = {},
                    onDrag = { change, offset -> change.consume() }
                )
            } else Modifier)
            .clip(RoundedCornerShape(20.dp)),
        shape = RoundedCornerShape(20.dp),
        color = MaterialTheme.colorScheme.surfaceVariant
    ) {
        Row(
            modifier = Modifier
                .clickable(onClick = onClick)
                .padding(horizontal = 12.dp, vertical = 6.dp),
            verticalAlignment = Alignment.CenterVertically
        ) {
            icon?.let {
                Icon(it, contentDescription = null, modifier = Modifier.size(16.dp),
                     tint = MaterialTheme.colorScheme.onSurface)
                Spacer(modifier = Modifier.width(4.dp))
            }
            Text(label, style = MaterialTheme.typography.labelMedium,
                 color = MaterialTheme.colorScheme.onSurface)

            if (editMode && onDelete != null) {
                Spacer(modifier = Modifier.width(4.dp))
                IconButton(onClick = onDelete, modifier = Modifier.size(14.dp)) {
                    Icon(Icons.Default.Close, contentDescription = "删除",
                         modifier = Modifier.size(10.dp), tint = Color.Red)
                }
            }
        }
    }
}
```

## 3.2 QuickActionRegistry 扩展

需要在 `QuickActionRegistry.kt`（`com.limelight` 包）新增操作，但**不修改原文件**。改为在 `com.alexclin.moonlink.stream.ui.common` 包下创建扩展：

```kotlin
// app/src/main/java/com/alexclin/moonlink/stream/ui/common/QuickActionExtensions.kt

object MoonLinkQuickActions {
    // 新增的 MoonLink 专属快捷操作 ID
    const val TOGGLE_PIP = "toggle_pip"
    const val TOGGLE_ADAPTIVE_BITRATE = "toggle_adaptive_bitrate"
    const val TOGGLE_CONTROL_ONLY = "toggle_control_only"
    const val TOGGLE_GYRO = "toggle_gyro"

    fun getAllActions(context: Context): List<QuickActionItem> {
        // 在 QuickActionRegistry 基础上追加新增操作
        val builtin = QuickActionRegistry.getAllActions(...)
        val extended = builtin.toMutableMap()
        extended[TOGGLE_PIP] = QuickActionItem(...)
        extended[TOGGLE_ADAPTIVE_BITRATE] = QuickActionItem(...)
        extended[TOGGLE_CONTROL_ONLY] = QuickActionItem(...)
        extended[TOGGLE_GYRO] = QuickActionItem(...)
        return extended
    }
}
```

### 操作执行分发

```kotlin
private fun executeAction(id: String, engine: StreamEngine) {
    when (id) {
        "send_win"     -> KeySender.sendWin(engine.conn)
        "toggle_audio" -> engine.toggleAudioMute()
        "toggle_hdr"   -> KeySender.sendHdrToggle(engine.conn)
        "toggle_mic"   -> engine.toggleMicrophoneButton()
        "send_sleep"   -> KeySender.sendSleep(engine.conn)
        "send_tab"     -> KeySender.sendTab(engine.conn)
        "send_alt_tab" -> KeySender.sendAltTab(engine.conn)
        "send_alt_f4"  -> KeySender.sendAltF4(engine.conn)
        "toggle_keyboard" -> engine.toggleKeyboard()
        "toggle_controller" -> engine.toggleVirtualController()
        "toggle_perf"  -> engine.togglePerformanceOverlay()
        "quit"         -> engine.disconnect()
        "quit_and_stop"-> engine.disconnectAndQuit()
        // MoonLink 新增
        "toggle_pip"        -> engine.togglePip()
        "toggle_adaptive_bitrate" -> engine.prefConfig.enableAdaptiveBitrate = ...
        "toggle_control_only" -> engine.prefConfig.controlOnly = ...
        "toggle_gyro"       -> engine.toggleGyro()
    }
}
```
