fun StreamOverlay(
    engine: StreamEngine,
    connectionStage: String? = null,
) {

    var activeEntry by remember { mutableStateOf<String?>(null) }

    val onToggle = {
        panelState = if (panelState == PanelState.HIDDEN) {
            PanelState.VERTICAL_BAR
        } else {
            PanelState.HIDDEN
        }
    }

    Box(modifier = Modifier.fillMaxSize()) {
        // 连接进度 overlay（阶段更新 + 首帧后自动消失）
        AnimatedVisibility(visible = connectionStage != null, ...) {
            Box(fillMaxSize, background(Color(0xBB000000)), Alignment.Center) {
                Text("启动中 ${connectionStage}")
            }
        }

        // 面板外点击关闭层（仅面板展开时）
        if (panelState != PanelState.HIDDEN) {

            visible = panelState == PanelState.HIDDEN,

            visible = panelState != PanelState.HIDDEN,
            enter = PanelAnimations.verticalBarEnter,
            exit = PanelAnimations.verticalBarExit,
            modifier = Modifier.align(
                if (isLandscape) Alignment.CenterEnd
                else Alignment.BottomCenter

## 1.2 FloatingActionButton.kt

**路径**：`app/src/main/java/com/alexclin/moonlink/stream/ui/FloatingActionButton.kt`

### 规格

| 属性 | 值 |
|------|------|
| 大小 | 36dp 圆形 |
| 透明度 | 100% |
| 图标 | `Icons.Default.Menu`（Material Symbols） |
| 阴影 | elevation 6dp |
| 背景色 | `primary`（`#00B4D8`） |
| 默认位置 | 横屏右上角 |
| 拖动行为 | 自由拖动，松手后停留在最后位置，不自动贴边 |

### 实现代码

```kotlin
@Composable
fun FloatingActionButton(
    visible: Boolean,
    onToggle: () -> Unit,
    modifier: Modifier = Modifier,
) {
    if (!visible) return

    val density = LocalDensity.current
    val configuration = LocalConfiguration.current
    val screenWidthPx = with(density) { configuration.screenWidthDp.dp.toPx() }
    val screenHeightPx = with(density) { configuration.screenHeightDp.dp.toPx() }
    val fabSizePx = with(density) { 36.dp.toPx() }

    var offsetX by remember { mutableFloatStateOf(screenWidthPx - fabSizePx) }
    var offsetY by remember { mutableFloatStateOf(0f) }

    Box(
        modifier = modifier
            .offset { IntOffset(offsetX.roundToInt(), offsetY.roundToInt()) }
            .size(36.dp)
            .shadow(6.dp, CircleShape)
            .background(MaterialTheme.colorScheme.primary, CircleShape)
            .pointerInput(Unit) {
                detectDragGestures { change, dragAmount ->
                    change.consume()
                    offsetX += dragAmount.x
                    offsetY += dragAmount.y
                    offsetX = offsetX.coerceIn(0f, screenWidthPx - fabSizePx)
                    offsetY = offsetY.coerceIn(0f, screenHeightPx - fabSizePx)
                }
            }
            .pointerInput(onToggle) {
                detectTapGestures(onTap = { onToggle() })
            },
        contentAlignment = Alignment.Center,
    ) {
        Icon(
            Icons.Default.Menu,
            contentDescription = "菜单",
            tint = MaterialTheme.colorScheme.onPrimary,
            modifier = Modifier.size(20.dp),
        )
    }
}
```

### 拖动逻辑说明

- 使用 `detectDragGestures` 实现拖动，无 `onDragEnd` 吸附逻辑
- 拖动范围限制在屏幕边界内（`coerceIn`）
- 点击使用独立的 `detectTapGestures`，与拖动互不干扰

## 1.3 VerticalBar.kt & HorizontalBar.kt

**路径**：`app/src/main/java/com/alexclin/moonlink/stream/ui/VerticalBar.kt`

### 布局规格

**VerticalBar（横屏右边缘）：**
```
┌──────────┐
│          │  ← 顶部留白（padding）
│   ⚙ 操   │  ← 操作按钮
│          │
│   ⌨ 键   │  ← 键盘按钮
│          │
│   🖥 桌   │  ← 显示桌面按钮
│          │
│   ⊞ 窗   │  ← 展示所有窗口按钮
│          │  ← 底部留白（padding）
└──────────┘
```

- 宽 60dp，左侧圆角 16dp，填满屏幕高度
- `Surface` + `shadowElevation = 8.dp`
- `Column` + `Arrangement.SpaceEvenly`

**HorizontalBar（竖屏底部）：**
```
┌──────────────────────────────────┐
│  ⚙ 操    ⌨ 键    🖥 桌    ⊞ 窗   │
└──────────────────────────────────┘
```

- 高 60dp，顶部圆角 16dp，填满屏幕宽度
- `Row` + `Arrangement.SpaceEvenly`

### 按钮组件（PanelEntry）

```kotlin
@Composable
internal fun PanelEntry(
    icon: ImageVector,
    label: String,
    isSelected: Boolean,
    onClick: () -> Unit,
) {
    // 背景/文字颜色使用 animateColorAsState 平滑过渡
    val bgColor by animateColorAsState(
        targetValue = if (isSelected) primaryContainer else surface,
    )
    val contentColor by animateColorAsState(
        targetValue = if (isSelected) onPrimaryContainer else onSurface,
    )

    Column(
        modifier = Modifier
            .fillMaxWidth()
            .then(if (isSelected) Modifier.background(bgColor, RoundedCornerShape(8.dp)) else Modifier)
            .clickable(onClick = onClick)
            .padding(vertical = 10.dp),
        horizontalAlignment = Alignment.CenterHorizontally,
    ) {
        Icon(icon, modifier = Modifier.size(22.dp), tint = contentColor)
        Spacer(modifier = Modifier.height(2.dp))
        Text(label, style = MaterialTheme.typography.labelSmall, color = contentColor)
    }
}
```

### 窄条切换逻辑

- 横屏（`ORIENTATION_LANDSCAPE`）：`VerticalBar`，`Alignment.CenterEnd`
- 竖屏：`HorizontalBar`，`Alignment.BottomCenter`
- 点击"操作"/"键盘"：toggle 高亮，切换到对应子面板状态
- 点击"桌面"/"窗口"：关闭面板（按键发送后续阶段实现）

## 1.4 PanelAnimations.kt

**路径**：`app/src/main/java/com/alexclin/moonlink/stream/ui/common/PanelAnimations.kt`

```kotlin
object PanelAnimations {
    /** 窄条进入（从右滑入） */
    val verticalBarEnter = slideInHorizontally(
        initialOffsetX = { it },
        animationSpec = tween(250, easing = FastOutSlowInEasing)
    ) + fadeIn(animationSpec = tween(200))

    /** 窄条退出（向右滑出） */
    val verticalBarExit = slideOutHorizontally(
        targetOffsetX = { it },
        animationSpec = tween(200, easing = FastOutSlowInEasing)
    ) + fadeOut(animationSpec = tween(150))

    /** 悬浮按钮出现 */
    val fabEnter = fadeIn(animationSpec = tween(200)) + scaleIn(
        initialScale = 0.5f,
        animationSpec = tween(200, easing = FastOutSlowInEasing)
    )

    /** 悬浮按钮消失 */
    val fabExit = fadeOut(animationSpec = tween(150)) + scaleOut(
        targetScale = 0.5f,
        animationSpec = tween(150)
    )

    /** 子面板进入（从右滑入） */
    val subPanelEnter = slideInHorizontally(
        initialOffsetX = { it },
        animationSpec = tween(250, easing = FastOutSlowInEasing)
    ) + fadeIn(animationSpec = tween(200))

    /** 子面板退出（向右滑出） */
    val subPanelExit = slideOutHorizontally(
        targetOffsetX = { it },
        animationSpec = tween(200, easing = FastOutSlowInEasing)
    ) + fadeOut(animationSpec = tween(150))
}
```

## 1.5 额外实现（阶段 1 新增，原设计未覆盖）

### 连接进度 overlay

- 文件：`StreamOverlay.kt` / `StreamActivity.kt`
- 半透明黑色遮罩 + 白色居中文字 "启动中 xxx"
- `connectionStage` 状态由 `StreamActivity` 管理，通过 `onStageUpdate` 回调更新
- 首帧解码时由 `firstFrameCallback` 发起 `onStageUpdate("connected", true, false)` 触发关闭
- `displayMessage` 回调显示 Toast

### 全屏沉浸式

- 文件：`StreamActivity.kt`
- `WindowCompat.setDecorFitsSystemWindows(window, false)` + `FLAG_FULLSCREEN` + `IMMERSIVE_STICKY`
- `onWindowFocusChanged` 中恢复沉浸式标志

### 生命周期：关屏/切应用退出串流

- `onPause` → `conn?.stop()`，重置 `connected`/`attemptedConnection`
- `onResume` → `recreate()` 重建 Activity 重新初始化

### 返回键退出串流

- `onBackPressed` → `conn?.stop()` 后 `finish()`
